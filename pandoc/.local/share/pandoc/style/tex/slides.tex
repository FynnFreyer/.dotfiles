\usefonttheme{default}

% Increase spacing between lines (verbatim is forced to 1 somewhere else)
\linespread{1.2}

\usepackage{etoolbox}

% Adjust spacing between lists and following blocks (e.g. verbatim)
\newlength{\afterlistsep}
\setlength{\afterlistsep}{.5em}  % Adjust the space here

% Command to add vertical space if at the outermost level
\makeatletter
\newcommand{\addoutermostlistvspace}[1]{%
  \ifnum\@listdepth<2
    \vspace{#1}
  \fi
}
\makeatother

% Use etoolbox to add vertical space after itemize/enumerate
\AtEndEnvironment{itemize}{\addoutermostlistvspace{\afterlistsep}}
\AtEndEnvironment{enumerate}{\addoutermostlistvspace{\afterlistsep}}

% Uncomment this to add vspace before itemize/enumerate as well
%\newlength{\beforelistsep}
%\setlength{\beforelistsep}{.5em}  % Adjust the space here

%\AtBeginEnvironment{itemize}{\addoutermostlistvspace{\beforelistsep}}
%\AtBeginEnvironment{enumerate}{\addoutermostlistvspace{\beforelistsep}}

% Set the font for quotes
\ifdef{\setbeamerfont}{
    % use small caps for quotes
    %\setbeamerfont{quote}{shape=\scshape}
}{}

% Set the quote coloring
\ifdef{\setbeamercolor}{
    \setbeamercolor{quote}{fg=black,bg=white}
}{}

\ifdef{\setbeamertemplate}{
    % spacing for quotes
    \setbeamertemplate{quote begin}{\vspace{15pt}}
    \setbeamertemplate{quote end}{\vspace{15pt}}

    % style bullets
    \setbeamertemplate{itemize items}[circle]
    \setbeamertemplate{enumerate items}[default]

    % add slide numbers to bottom
    \setbeamertemplate{footline}{
        \hfill
        \insertframenumber/\inserttotalframenumber
        \hspace{5pt}\vspace{5pt}
    }
}{}
